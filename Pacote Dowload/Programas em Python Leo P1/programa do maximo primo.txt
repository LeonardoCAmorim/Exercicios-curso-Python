def éPrimo(k):
    if k < 2:
        return False
    elif k == 2 or k == 3 or k == 5 or k == 7:
        return True
    else:
      if k % 2 == 0 or k % 3 == 0 or k % 5 == 0 or k % 7 == 0:
        return False
      i = 11
      while i * i <= k:
        if k % i == 0 or k % (i + 2) == 0:
          return False
        i += 6
      return True
          
def maior_primo(a):
    while a >= 2:
        if éPrimo(a):
            return a
        a = a - 1
