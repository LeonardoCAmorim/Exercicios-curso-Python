def menor_nome(nomes):
  
    nomes_Certo = []
    for nome in nomes:
        nomes_Certo.append(nome.strip().capitalize())
        
    i = 1
    menor_nome = nomes_Certo[0]
    while i < len(nomes_Certo):
        if len(nomes_Certo[i]) < len(menor_nome):
            menor_nome = nomes_Certo[i]
        i += 1
      
    return menor_nome
