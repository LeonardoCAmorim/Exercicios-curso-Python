    aux = limite_de_pecas
    deu_certo_essa_bagaça_maluca = 0
  
    while aux > 0:
        if (qtd_pecas - aux) % (limite_de_pecas + 1) == 0:
            deu_certo_essa_bagaça_maluca = True
            return aux
        else:
            aux = aux - 1
          
    if deu_certo_essa_bagaça_maluca == False:
        return limite_de_pecas
